    
    LD R0, 60
    LD R1, 2
    LD R2, 20

    MOV ACC, R0  # Seta 120 para R0
    ADD ACC, R0
    MOV R0, ACC

loop1:
    MOV ACC, R0
    SW R1  # Salva o valor do Acumulador no endereço de memória de R1

    MOV ACC, R2  # Atualiza o índice
    SUB ACC, R1
    MOV R2, ACC

    MOV ACC, R0  # Incrementa o número
    ADD ACC, R1
    MOV R0, ACC

    MOV ACC, R2
    CMP R1
    BGT loop1

    LW R0  # Carrega o valor no endereco de memoria de R0 no Acumulador
